create table examresults
(sid int not null,
marks int
);

select * from examresults;
select * from student inner join examresults where student.sid=examresults.sid;
select * from student natural join examresults where student.sid=examresults.sid;
select * from student left outer join examresults on student.sid=examresults.sid;
select * from student right outer join examresults on student.sid=examresults.sid;
