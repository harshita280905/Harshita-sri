#group by
select department, count(department) from 
worker group by department;

select department, MAX(salary) AS max_salary
from worker
group by department;

select department, MIN(salary) AS min_salary
from worker
group by department;

#group by and order by

order by department asc;

#offset
SELECT *

#offset and fetch
SELECT * 
FROM Worker

#limit

FROM Worker

#limit and offset
SELECT *

#not in with subquery
SELECT * 
FROM Worker
WHERE WORKER_ID NOT IN (
    SELECT WORKER_ID 
    FROM Worker
    ORDER BY WORKER_ID
    LIMIT 5
);

#row number
SELECT WORKER_ID, FIRST_NAME, LAST_NAME, SALARY, DEPARTMENT,
       ROW_NUMBER() OVER (ORDER BY WORKER_ID) AS RowNum
FROM Worker;

SELECT *
FROM (
    SELECT *, ROW_NUMBER() OVER (ORDER BY WORKER_ID) AS rn
    FROM Worker
